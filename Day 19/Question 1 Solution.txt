class Tree
{
    boolean flag = true;
    
    int postOrder(Node root){
        if(root==null)
            return 0;
        if(root.left==null && root.right==null)
            return 1;
        int left = postOrder(root.left);
        int right = postOrder(root.right);
        int diff = left-right;
        if(diff<-1 || diff>1)
            flag = false;
        int max = (left>right)?left:right;
        return max+1;
    }
    
    boolean isBalanced(Node root)
    {
	    postOrder(root);
	    return flag;
    }
}
