class Solution
{
    static int[] dijkstra(ArrayList<ArrayList<Integer>> g, int src, int V)
    {
        int[] result = new int[V];
        HashSet<Integer> visited = new HashSet<>(V);
        for(int i=0;i<V;++i){
            result[i]=Integer.MAX_VALUE;
        }
        result[src]=0;
        int min = 0;
        int focus = src;
        while(min!=Integer.MAX_VALUE){
            visited.add(focus);
            ArrayList<Integer> neighbours = g.get(focus);
            int index = focus;
            min = Integer.MAX_VALUE;
            for(int i=0;i<neighbours.size();++i){
                if(visited.contains(i))
                    continue;
                if(!neighbours.get(i).equals(0) && result[focus]+neighbours.get(i)<result[i]){
                    result[i] = result[focus]+neighbours.get(i);
                }
                if(result[i]<min){
                    min = result[i];
                    index = i;
                }
            }
            focus = index;
        }
        return result;
    }
}
